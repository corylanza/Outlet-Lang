class node {

	int Val;
	node Next;

	node(int val, node next) {
		Val = val;
		Next = next;
	}
}
class list {

	node Head = null;

	void addfront(int v) {
		Head = node(v, Head);
	}

	void printlist() {
		node cur = Head;
		while(cur != null) {
			console.print(cur.Val);
			cur = cur.Next;
		}
	}
}

list l = list();
l.addfront(3);
l.addfront(6);
l.addfront(9);
l.printlist();
