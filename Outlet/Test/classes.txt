class person {
	string Name;
	int Age = 0;
	static int count = 0;
	
	void person(string n, int a) {
		Name = n;
		Age = a;
		make();
		print("contructed "+Name+" "+Age+" count "+count);
	}

	(string, int) deconstruct() => (Name, Age);

	void greet() => print("hello i am " + Name);

	static void make() => count++;
}

class math {
	static int fac(int n) => n == 1 ? 1 : n*fac(n-1);
	static int pow(int b, int e) => e == 0 ? 1 : b*pow(b, e-1);
}
print(math.fac(5));
print(math.pow(10,3));
