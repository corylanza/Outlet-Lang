class person {
	string Name;
	int Age = 0;
	static int count = 0;
	
	person(string n, int a) {
		Name = n;
		Age = a;
		make();
		print("contructed "+Name+" "+Age+" count "+count);
	}

	(string, int) deconstruct() => (Name, Age);

	void greet() => print("hello i am " + Name);
	
	static void make() { count++; }
}

class std {
	string ha;
}


print(person.count);
person a = person("cory", 20);
person b = person("jenna", 21);
a.greet();
b.greet();
print(a.deconstruct());
print(person.count);
std c = std();
c.ha = "hahaha";
print(c.ha);
