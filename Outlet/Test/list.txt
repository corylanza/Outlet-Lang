class Node {

	int Value;
	Node Next;
	type this = Node;

	void Node(int value, this next) => {
		Value = value;
		Next = next;
	}
}

class List {

	Node Head;
	Node Tail;

	void List(int a) => {
		Head = Node(a, null);
		Tail = Head;
	}

	void add(int a) => {
		Node temp = Node(a, null);
		Tail.Next = temp;
		Tail = temp;
	}

	void printlist() => {
		Node cur = Head;
		while(cur != null){
			print(cur.Value);
			cur = cur.Next;
		}
	}
}

List a = List(5);
a.printlist();
