class person {

	static int count = 0;
	static void make() => console.print("made");

	string Name;
	int Age;

	person(string name, int age) {
		count++;
		this.Name = name;
		Age = age;
	}
	static void greet() => console.print("hello!");
	void intro() => console.print("Hello, my name is "+Name);
}
